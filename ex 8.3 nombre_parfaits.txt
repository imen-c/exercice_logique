debut nombreparfait
DEBUT VARIABLES LOCALES
VAR somme  :N <- 0
VAR nombre :N <- 0
VAR nombreinferieur :N <- 0
VAR limite :N <- 100
FIN VARIABLES LOCALES



   POUR nombre  ALLANT DE 2 A (limite-1)
        POUR nombreinferieur ALLANT DE 1 A (nombre-1)
		        SI nombreMODnombreinferieur = 0
		        alors
		        somme<-somme + nombreinferieur
		        FINSI
        nombreinferieur SUIVANT
		SI somme = nombre
		alors 
		     AFFICHE nombre + "est un nombre parfait"
		FINSI 	
   nombre SUIVANT

fin nombreparfait
